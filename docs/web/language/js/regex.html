    # 取出版本号

    "Ubuntu 8.10"，取出主版本号8和次版本号10

    let regexp = /\b\d+(?=\.)/
regexp.exec('Ubuntu 8.10')
// [0: "8"
// groups: undefined
// index: 7
// input: "Ubuntu 8.10"]
let regexp = /(?<=\.)\d+/
regexp.exec('Ubuntu 8.10')
// [0: "10"
// groups: undefined
// index: 9
// input: "Ubuntu 8.10"
// length: 1]
