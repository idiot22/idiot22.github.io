    微任务队列queueJob

    二喵 3/3/2022

    effect函数要是不做限制，要是触发的函数有多个值同时变化，或者一个值变化多次，就会导致effect触发多次，很耗性能

    # 场景

    如果没有通过手段进行管理，那么会更新100次，但是只有当 for 被执行完成之后更新才是有必要的，Vue3中就是利用微任务队列的特点来进行统一管理的

    const count = ref(0)
 
<template>
<div>{{ count }}</div>

for(let i = 0; i < 100; i++) {
  count ++

    # 代码实现

    let queue = []
export function queueJob(job){
  if(job && !queue.includes(job)){ // 相同的任务就不放到任务队列中去
    queue.push(job)
    queueFlush()
  }
}
let isFlushPending = false
function queueFlush(){
  if(!isFlushPending){
    isFlushPending = true  // 正在执行队列就不执行新的任务
    // 用promise包裹就将执行任务变成微任务，不会立即执行
    // 会等到前面的同步的任务执行完成，就是变化的值对应的effects全部forEach完后才会调用
    // 这样就可以确保queue里面是所有的effects且不重复
    Promise.resolve().then(flushJobs)
  }
}
function flushJobs(){
  isFlushPending = false
  // 清空时，我们需要根据调用顺序依次刷新，保证先刷新父再刷新子
  queue.sort((a,b) => a.id - b.id)
  for(let i=0;i< queue.length;i++){
    const job = queue[i]
    if(job){
      job()
    }
  }
  console.log('flushJobs', queue, 888)
  queue.length = 0
}
