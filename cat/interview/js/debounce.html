    防抖，节流

    二喵 1/18/2022

    # 防抖，节流定义

    # 防抖（debounce）

    n 秒后在执行该事件，若在 n 秒内被重复触发，则重新计时

    # 节流（throttle）

    n 秒内只运行一次，若在 n 秒内重复触发，只有一次生效

    # 代码

    防抖

    // 不立即执行
function debounce(fn, wait){
 let timeout
 return function(...args){
   let context = this
   if(timeout){
     clearTimeOut(timeout)
   }
   timeout = setTimeOut(()=>{
     fn.apply(context, args)
   }, wait)
 }
}

    // 立即执行
function debounce(fn, wait, immediate){
  let timeout
  return function(...args){
    let context = this
    if(timeout){
      clearTimeOut(timeout) // timeout不为null
    }
    if(immediate){
      let callNow = !timeout
      timeout = setTimeOut(() => {
        timeout = null
      }, wait)
      if(callNow){
        fn.apply(context, args)
      }
    }else{
      timeout = setTimeOut(() => {
        fn.apply(context, args)
      }, wait)
    }
  }
}

    节流

    function throttle(fn, wait){
  let timeout
  let startTime = Date.now()
  return function(...args){
    let context = this
    let currentTime = Date.now
    if(timeout){
       clearTimeOut(timeout) // timeout不为null
    }
    if(currentTime - startTime > wait){
      fn.apply(context, args)
      startTime = Date.now()
    }else{
      timeout = setTimeOut(()=>{
       fn.apply(context, args)
      }, wait)
    }
  }
}
