    function MyNew(Func, ...args){
  // 新建一个对象
  let obj = {}
  // 将对象的__proto__赋值Func.protoType
  obj.__proto__ = Func.protoType
  // 将func中的属性赋值到obj
  let result = Func.apply(obj, ...args)
  // 判断Func返回值，如果是obj就返回result，不是就返回obj
  return typeof result === 'object' ? result : obj
}
